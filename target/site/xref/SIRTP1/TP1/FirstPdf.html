60      private static void addTitlePage(Document document)
61              throws DocumentException {
62          Paragraph preface = new Paragraph();
63          // We add one empty line
64          addEmptyLine(preface, 1);
65          // Lets write a big header
66          preface.add(new Paragraph("Title of the document", catFont));
67  
68          addEmptyLine(preface, 1);
69          // Will create: Report generated by: _name, _date
70          preface.add(new Paragraph(
71                  "Report generated by: " + System.getProperty("user.name") + ", " + new Date(), //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
72                  smallBold));
73          addEmptyLine(preface, 3);
74          preface.add(new Paragraph(
75                  "This document describes something which is very important ",
76                  smallBold));
77  
78          addEmptyLine(preface, 8);
79  
80          preface.add(new Paragraph(
81                  "This document is a preliminary version and not subject to your license agreement or any other agreement  ;-).",
82                  redFont));
83  
84          document.add(preface);
85          // Start a new page
86          document.newPage();
87      }
